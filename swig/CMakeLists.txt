########################################################################
# Check if there is C++ code at all
########################################################################
if(NOT qpsk_sources)
	MESSAGE(STATUS "No C++ sources... skipping swig/")
	return()
endif(NOT qpsk_sources)

########################################################################
# Include swig generation macros
########################################################################
find_package(SWIG)
find_package(PythonLibs 2)
if(NOT SWIG_FOUND OR NOT PYTHONLIBS_FOUND)
    message(STATUS "SWIG or Python libs not found. Skipping Python support.")
    return()
endif()
include(GrSwig)
include(GrPython)

########################################################################
# Setup swig generation
########################################################################
set(GR_SWIG_INCLUDE_DIRS
    $<TARGET_PROPERTY:gnuradio::runtime_swig,INTERFACE_INCLUDE_DIRECTORIES>
    ${UHD_INCLUDE_DIRS}
)
set(GR_SWIG_TARGET_DEPS gnuradio::runtime_swig)

set(GR_SWIG_LIBRARIES gnuradio-qpsk)
set(GR_SWIG_DOC_FILE ${CMAKE_CURRENT_BINARY_DIR}/qpsk_swig_doc.i)
set(GR_SWIG_DOC_DIRS ${CMAKE_CURRENT_SOURCE_DIR}/../include)

GR_SWIG_MAKE(qpsk_swig qpsk_swig.i)

########################################################################
# Install the build swig module
########################################################################
GR_SWIG_INSTALL(TARGETS qpsk_swig DESTINATION ${GR_PYTHON_DIR}/qpsk)

########################################################################
# Install swig .i files for development
########################################################################
install(
    FILES
    qpsk_swig.i
    ${CMAKE_CURRENT_BINARY_DIR}/qpsk_swig_doc.i
    DESTINATION ${GR_INCLUDE_DIR}/qpsk/swig
)
